  if(n<1000) return n<10?n.toFixed(2).replace(/\.?0+$/,''):Math.floor(n).toString();

  while(n>=1000 && i<units.length-1){n/=1000;i++}
  return n.toFixed(2).replace(/\.?0+$/,'')+units[i];
}
function upgradeCost(u, lvl){
  return Math.floor(u.baseCost * Math.pow(u.costMul, lvl));
}
function getLvl(id){return STATE.upgrades[id]||0}

function calcStats(){
  let clickAdd = 1; // base click
  let cpsAdd = 0;
  let clickMult = 1;
  let cpsMult = 1;
  let mult = 1;
  let starBonusAdd = 0;
  let cosmicBonusMul = 1;
  let autoClick = 0;
  let offlineBoost = 0.1; // base 10% offline

  for(const u of UPGRADES){
    const lvl = getLvl(u.id);
    if(lvl<=0) continue;
    const e = u.effect(lvl);
    if(e.type==='clickAdd') clickAdd += e.val;
    else if(e.type==='cpsAdd') cpsAdd += e.val;
    else if(e.type==='clickMult') clickMult *= e.val;
    else if(e.type==='cpsMult') cpsMult *= e.val;
    else if(e.type==='mult') mult *= e.val;
    else if(e.type==='starBonus') starBonusAdd += e.val;
    else if(e.type==='cosmicBonus') cosmicBonusMul *= e.val;
    else if(e.type==='autoClick') autoClick += e.val;
    else if(e.type==='offlineBoost') offlineBoost = Math.max(offlineBoost, e.val);
  }
  // Currency multipliers
  const starMul = 1 + STATE.stars * 0.05;
  const cosmicMul = 1 + STATE.cosmic * 0.5;
  const voidMul = 1 + STATE.voidEssence * 1.0;

  const finalMult = mult * starMul * cosmicMul * voidMul;
  const perClick = (clickAdd * clickMult) * finalMult;
  const perSec = cpsAdd * cpsMult * finalMult;

  return {perClick, perSec, mult: finalMult, autoClick, starBonusAdd, cosmicBonusMul, offlineBoost};
}

// Star gain formula based on total coins earned this prestige run
function calcStarGain(){
  const stats = calcStats();
  if(STATE.totalCoinsEarned < 1e4) return 0;
  // Cube root scaling
  const base = Math.floor(Math.pow(STATE.totalCoinsEarned/1e4, 0.5));
  return base + stats.starBonusAdd * (base>0?1:0);
}
function calcCosmicGain(){
  const stats = calcStats();
  if(STATE.totalStars < 50) return 0;
  const base = Math.floor(Math.pow(STATE.totalStars/50, 0.6));
  return Math.floor(base * stats.cosmicBonusMul);
}
function calcVoidGain(){
  if(STATE.totalCosmic < 25) return 0;
  return Math.floor(Math.pow(STATE.totalCosmic/25, 0.5));
}

// ======= CLICK =======
function doClick(x, y){
  const stats = calcStats();
  const gain = stats.perClick;
  STATE.coins += gain;
  STATE.totalCoinsEarned += gain;
  STATE.totalClicks++;
  spawnFloat(x, y, '+'+fmt(gain));
  spawnParticles(x, y);
  checkAchievements();
}

function spawnFloat(x, y, txt){
  const el = document.createElement('div');
  el.className = 'float';
  el.textContent = txt;
  el.style.left = (x-20) + 'px';
  el.style.top = (y-20) + 'px';
  document.querySelector('.center').appendChild(el);
  setTimeout(()=>el.remove(), 1000);
}
function spawnParticles(x, y){
  const c = document.querySelector('.center');
  for(let i=0;i<6;i++){
    const p = document.createElement('div');
    p.className='particle';
    const a = (Math.PI*2*i)/6 + Math.random()*.5;
    const d = 40 + Math.random()*30;
    p.style.setProperty('--dx', Math.cos(a)*d+'px');
    p.style.setProperty('--dy', Math.sin(a)*d+'px');
    p.style.left = x+'px';
    p.style.top = y+'px';
    if(STATE.transcends>0) p.style.background = 'var(--cyan)';
    else if(STATE.ascensions>0) p.style.background = 'var(--magenta)';
    c.appendChild(p);
    setTimeout(()=>p.remove(), 600);
  }
}

document.getElementById('clicker').addEventListener('click', e=>{
  const center = document.querySelector('.center').getBoundingClientRect();
  const x = e.clientX - center.left;
  const y = e.clientY - center.top;
  doClick(x, y);
});

// ======= UPGRADE BUYING =======
function buyUpgrade(id){
  const u = UPGRADES.find(x=>x.id===id);
  if(!u) return;
  const lvl = getLvl(id);
  if(lvl >= u.max) return;
  const cost = upgradeCost(u, lvl);
  const cur = u.currency || 'coins';
  if(STATE[cur] < cost) return;
  STATE[cur] -= cost;
  STATE.upgrades[id] = lvl + 1;
  log(`Bought ${u.name} Lv ${lvl+1}`, cur==='coins'?'':cur==='stars'?'magenta':'cyan');
  renderUpgrades();
}

// ======= PRESTIGE / ASCEND / TRANSCEND =======
function doPrestige(){
  const gain = calcStarGain();
  if(gain<=0) return;
  STATE.stars += gain;
  STATE.totalStars += gain;
  STATE.prestiges++;
  // reset coins and basic upgrades
  STATE.coins = 0;
  STATE.totalCoinsEarned = 0;
  for(const u of UPGRADES){
    if(u.tier === 'basic') delete STATE.upgrades[u.id];
  }
  log(`★ PRESTIGED for ${gain} Stars`, 'magenta');
  showToast('PRESTIGE', `Gained ${gain} ★`);
  STATE.currentTab = 'star';
  renderAll();
  checkAchievements();
}
function doAscend(){
  const gain = calcCosmicGain();
  if(gain<=0) return;
  STATE.cosmic += gain;
  STATE.totalCosmic += gain;
  STATE.ascensions++;
  STATE.coins = 0;
  STATE.totalCoinsEarned = 0;
  STATE.stars = 0;
  STATE.totalStars = 0;
  for(const u of UPGRADES){
    if(u.tier === 'basic' || u.tier === 'star') delete STATE.upgrades[u.id];
  }
  log(`◇ ASCENDED for ${gain} Cosmic Energy`, 'cyan');
  showToast('ASCEND', `Gained ${gain} ◇`);
  STATE.currentTab = 'cosmic';
  renderAll();
  checkAchievements();
}
function doTranscend(){
  const gain = calcVoidGain();
  if(gain<=0) return;
  STATE.voidEssence += gain;
  STATE.transcends++;
  // Reset everything except void essence and transcends
  STATE.coins = 0;
  STATE.totalCoinsEarned = 0;
  STATE.stars = 0;
  STATE.totalStars = 0;
  STATE.cosmic = 0;
  STATE.totalCosmic = 0;
  STATE.upgrades = {};
  STATE.totalClicks = 0;
  log(`∞ TRANSCENDED for ${gain} Void Essence`, 'cyan');
  showToast('TRANSCEND', `Gained ${gain} ∞ — Reality reshaped`);
  STATE.currentTab = 'basic';
  renderAll();
  checkAchievements();
}

document.getElementById('btnPrestige').onclick = doPrestige;
document.getElementById('btnAscend').onclick = doAscend;
document.getElementById('btnVoid').onclick = doTranscend;

// ======= ACHIEVEMENTS =======
function checkAchievements(){
  for(const a of ACHIEVEMENTS){
    if(STATE.achievements[a.id]) continue;
    if(a.cond(STATE)){
      STATE.achievements[a.id] = true;
      showToast('ACHIEVEMENT', a.name);
      log(`◆ Achievement: ${a.name}`, 'gold');
    }
  }
}

// ======= LOG =======
const logEl = document.getElementById('log');
function log(msg, cls=''){
  const e = document.createElement('div');
  e.className = 'e ' + cls;
  const t = new Date();
  const ts = String(t.getHours()).padStart(2,'0')+':'+String(t.getMinutes()).padStart(2,'0')+':'+String(t.getSeconds()).padStart(2,'0');
  e.innerHTML = `<span class="t">${ts}</span>${msg}`;
  logEl.insertBefore(e, logEl.firstChild);
  while(logEl.children.length > 40) logEl.lastChild.remove();
}

// ======= RENDER =======
function renderUpgrades(){
  const list = document.getElementById('upgradeList');
  list.innerHTML = '';
  const tier = STATE.currentTab;
  const filtered = UPGRADES.filter(u=>u.tier===tier);
  for(const u of filtered){
    const lvl = getLvl(u.id);
    const cost = upgradeCost(u, lvl);
    const cur = u.currency || 'coins';
    const have = STATE[cur];
    const can = have >= cost && lvl < u.max;
    const maxed = lvl >= u.max;

    const div = document.createElement('div');
    div.className = 'upgrade ' + (maxed?'maxed':can?'affordable':'');
    div.innerHTML = `
      <div class="up-row">
        <div class="up-name"><span class="up-icon">${u.icon}</span>${u.name}</div>
        <div class="up-lvl">Lv ${lvl}${u.max?'/'+u.max:''}</div>
      </div>
      <div class="up-desc">${u.desc}</div>
      <div class="up-cost ${can?'':'cant'}">${maxed?'✓ MAXED':`${fmt(cost)} ${cur==='coins'?'coins':cur==='stars'?'★':'◇'}`}</div>
    `;
    if(!maxed) div.onclick = ()=>{ buyUpgrade(u.id); renderHeader(); };
    list.appendChild(div);
  }
}

// ======= GAME LOOP =======
let autoClickAccum = 0;
function tick(){
  const now = Date.now();
  const dt = (now - STATE.lastTick) / 1000;
  STATE.lastTick = now;

  const stats = calcStats();
  const earned = stats.perSec * dt;
  STATE.coins += earned;
  STATE.totalCoinsEarned += earned;

  // Auto-click
  if(stats.autoClick > 0){
    autoClickAccum += stats.autoClick * dt;
    while(autoClickAccum >= 1){
      STATE.coins += stats.perClick;
      STATE.totalCoinsEarned += stats.perClick;
      STATE.totalClicks++;
      autoClickAccum -= 1;
    }
  }

  renderHeader();
  checkAchievements();
}
setInterval(tick, 100);

const SAVE_KEY = 'cosmic-clicker-save';
function save(){
  try{
    // We use in-memory + sessionStorage stand-in via window since localStorage may be restricted
    window.__save = JSON.stringify({...STATE, lastTick: Date.now()});
    showToast('SAVE', 'Game saved to session');
    log('◆ Game saved', 'gold');
  }catch(e){console.error(e)}
}
function load(){
  try{
    if(window.__save){
      const data = JSON.parse(window.__save);
      Object.assign(STATE, data);
      // Offline gain
      const offline = (Date.now() - STATE.lastTick)/1000;
      if(offline > 5){
        const stats = calcStats();
        const earned = stats.perSec * offline * stats.offlineBoost;
        STATE.coins += earned;
        STATE.totalCoinsEarned += earned;
        if(earned>0) showToast('OFFLINE', `+${fmt(earned)} coins (${Math.floor(offline)}s away)`);
      }
      STATE.lastTick = Date.now();
    }
  }catch(e){console.error(e)}
}
